DO $$
DECLARE
    t RECORD;
    c RECORD;
BEGIN
    -- Loop over all tables in your schema
    FOR t IN
        SELECT table_name
        FROM information_schema.tables
        WHERE table_schema='public' AND table_type='BASE TABLE'
    LOOP
        -- Loop over all text-like columns that are NOT generated
        FOR c IN
            SELECT column_name, data_type
            FROM information_schema.columns
            WHERE table_name = t.table_name
              AND data_type IN ('text', 'varchar', 'character varying')
              AND is_generated = 'NEVER'
        LOOP
            -- Update each column replacing localhost:9000 with LAN IP
            EXECUTE format(
                'UPDATE %I SET %I = REPLACE(%I, %L, %L) WHERE %I LIKE %L',
                t.table_name,
                c.column_name,
                c.column_name,
                'localhost:9000',
                '192.168.0.232:9000',
                c.column_name,
                'http://localhost:9000/%'
            );
        END LOOP;
    END LOOP;
END $$;
